futures_list - хранилище тикетов фьючи и акции (ручная сборка)
CREATE TABLE "futures_list" (
        "futures"       TEXT,
        "paper" TEXT
);
sqlite>

futures_oi - исторические значения открытого интереса разбитые на - даты, инструмент,  
лица (юридические и физические), покупки и продажи.
CREATE TABLE futures_oi (
    id INTEGER PRIMARY KEY AUTOINCREMENT,

    futures TEXT,          -- код из futures_list
    paper TEXT,            -- человекочитаемое имя

    tradedate DATE,        -- дата

    FIZ_long INTEGER,
    FIZ_short INTEGER,

    YUR_long INTEGER,
    YUR_short INTEGER,

    UNIQUE(futures, tradedate)  -- защита от дублей
);

paper_price - исторические данные цен акций
CREATE TABLE "paper_price" (
        "id"    INTEGER,
        "ticker"        TEXT,
        "date"  DATE,
        "open"  REAL,
        "close" REAL,
        "high"  REAL,
        "low"   REAL,
        "volume"        INTEGER,
        PRIMARY KEY("id" AUTOINCREMENT)
);
CREATE UNIQUE INDEX idx_unique_ticker_date
ON paper_price (ticker, date);
